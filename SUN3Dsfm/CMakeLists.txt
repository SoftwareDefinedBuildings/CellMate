CMAKE_MINIMUM_REQUIRED(VERSION 2.6.0)
PROJECT(BA2D3Dbox CXX)

FIND_PACKAGE(Ceres REQUIRED)

ADD_EXECUTABLE(ba2D3Dbox ba2D3Dbox.cc)
TARGET_LINK_LIBRARIES(ba2D3Dbox ${CERES_LIBRARIES})

ADD_EXECUTABLE(ba2D3DboxType ba2D3DboxType.cc)
TARGET_LINK_LIBRARIES(ba2D3DboxType ${CERES_LIBRARIES})

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -pthread")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -pthread")
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -pthread")
